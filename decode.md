05 00 0e 02 04 04

as mouse
      0e 02 04 04  <- enable mouse
      0e 02 04 01  <- disable
      0e 05 04 01  <- always on
      0e 05 04 02 18 <- button rb
      0e 05 04 02 16 19 <- lb + rt
      0e 05 04 02 19    <- rt (hold to activate)
      0e 05 04 03 19    <- rt (toggle)
      0e 04 04 00 00 01 01      <- sensitivity 1
      0e 04 04 00 00 64 64      <- sensitivity 100
      0e 04 04 00 00 32 32      <- sensitivity 50
      0e 04 04 00 00 0a 0a      <- sensitivity 10
      0e 04 04 00 00 00 00 02 01    <- invert x on
      0e 04 04 00 00 00 00 01 01    <- invert x off
      0e 04 04 00 00 00 00 01 02    <- invert y on
      0e 04 04 00 00 00 00 01 01    <- invert y off

as l stick
      0e 02 04 02  <- enable lstick
      0e 02 04 03  <- enable rstick
      0e 03 04 00 00 01 00 00 ff ff ff ff <- instant
      0e 03 04 00 00 02 00 00 ff ff ff ff <- continous
      0e 05 04 02 19 00 00 00             <- hold to activate
      0e 05 04 03 19 00 00 00             <- toggle
      0e 03 04 00 00 00 00 00 ff ff ff ff 02 01 <- inert x on
      0e 03 04 00 00 00 32 32 ff ff ff ff       <- sensitivity 50
